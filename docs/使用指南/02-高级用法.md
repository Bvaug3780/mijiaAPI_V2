# 高级用法

## 多用户场景

### 场景说明
在多用户平台（如智能家居管理平台）中，需要同时管理多个用户的设备。

### 实现方式

```python
from mijiaAPI_V2 import create_multi_user_clients
from mijiaAPI_V2.infrastructure.credential_store import FileCredentialStore

# 1. 加载多个用户的凭据
store = FileCredentialStore()
credentials = {
    "user_a": store.load("user_a_credential.json"),
    "user_b": store.load("user_b_credential.json"),
    "user_c": store.load("user_c_credential.json"),
}

# 2. 创建多用户客户端
clients = create_multi_user_clients(credentials)

# 3. 为每个用户执行操作
for user_id, client in clients.items():
    homes = client.get_homes()
    print(f"用户 {user_id} 有 {len(homes)} 个家庭")
    
    for home in homes:
        devices = client.get_devices(home_id=home.id)
        print(f"  家庭 {home.name} 有 {len(devices)} 个设备")
```

### 缓存隔离

**为什么需要缓存隔离？**
- 不同用户的数据不应该混淆
- 避免缓存污染

**实现原理**：
```python
# 每个用户使用独立的缓存命名空间
cache_key = f"devices:{home_id}"
namespace = credential.user_id  # 使用用户ID作为命名空间

# 用户A的缓存
cache.set(cache_key, devices_a, namespace="user_a")

# 用户B的缓存（完全独立）
cache.set(cache_key, devices_b, namespace="user_b")
```

### 并发处理

```python
import concurrent.futures

def get_user_devices(user_id, client):
    """获取单个用户的设备"""
    try:
        homes = client.get_homes()
        all_devices = []
        for home in homes:
            devices = client.get_devices(home_id=home.id)
            all_devices.extend(devices)
        return user_id, all_devices
    except Exception as e:
        return user_id, None

# 并发获取所有用户的设备
with concurrent.futures.ThreadPoolExecutor(max_workers=5) as executor:
    futures = [
        executor.submit(get_user_devices, user_id, client)
        for user_id, client in clients.items()
    ]
    
    for future in concurrent.futures.as_completed(futures):
        user_id, devices = future.result()
        if devices:
            print(f"用户 {user_id}: {len(devices)} 个设备")
        else:
            print(f"用户 {user_id}: 获取失败")
```

## 异步API

### 为什么使用异步？

**适用场景**：
- Web应用（FastAPI、aiohttp等）
- 需要并发控制多个设备
- 高性能要求的场景

**性能对比**：
```python
# 同步：顺序执行，总时间 = 每个操作时间之和
devices_1 = api.get_devices(home_id_1)  # 1秒
devices_2 = api.get_devices(home_id_2)  # 1秒
devices_3 = api.get_devices(home_id_3)  # 1秒
# 总时间：3秒

# 异步：并发执行，总时间 ≈ 最慢操作的时间
devices_1, devices_2, devices_3 = await asyncio.gather(
    api.get_devices(home_id_1),
    api.get_devices(home_id_2),
    api.get_devices(home_id_3),
)
# 总时间：约1秒
```

### 基本使用

```python
import asyncio
from mijiaAPI_V2 import create_async_api_client

async def main():
    # 创建异步客户端
    api = create_async_api_client(credential)
    
    # 获取家庭列表
    homes = await api.get_homes()
    print(f"找到 {len(homes)} 个家庭")
    
    # 获取设备列表
    devices = await api.get_devices(home_id=homes[0].id)
    print(f"找到 {len(devices)} 个设备")
    
    # 控制设备
    success = await api.control_device(
        device_id=devices[0].did,
        siid=2,
        piid=1,
        value=True
    )
    print(f"控制结果: {success}")

# 运行异步函数
asyncio.run(main())
```

### 并发控制多个设备

```python
async def control_multiple_devices():
    api = create_async_api_client(credential)
    
    # 并发控制多个设备
    results = await asyncio.gather(
        api.control_device("device_1", 2, 1, True),
        api.control_device("device_2", 2, 1, False),
        api.control_device("device_3", 2, 2, 50),
        return_exceptions=True  # 不因单个失败而中断
    )
    
    for i, result in enumerate(results, 1):
        if isinstance(result, Exception):
            print(f"设备{i}控制失败: {result}")
        else:
            print(f"设备{i}控制成功: {result}")

asyncio.run(control_multiple_devices())
```

### 在Web框架中使用

**FastAPI示例**：
```python
from fastapi import FastAPI, HTTPException
from mijiaAPI_V2 import create_async_api_client

app = FastAPI()

# 全局API客户端（实际应该根据用户创建）
api = create_async_api_client(credential)

@app.get("/homes")
async def get_homes():
    """获取家庭列表"""
    try:
        homes = await api.get_homes()
        return {"homes": [h.model_dump() for h in homes]}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/devices/{home_id}")
async def get_devices(home_id: str):
    """获取设备列表"""
    try:
        devices = await api.get_devices(home_id=home_id)
        return {"devices": [d.model_dump() for d in devices]}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.post("/devices/{device_id}/control")
async def control_device(device_id: str, siid: int, piid: int, value: any):
    """控制设备"""
    try:
        success = await api.control_device(device_id, siid, piid, value)
        return {"success": success}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
```

## 批量操作

### 批量控制设备

```python
# 准备批量请求
requests = [
    {"device_id": "device_1", "siid": 2, "piid": 1, "value": True},
    {"device_id": "device_2", "siid": 2, "piid": 1, "value": False},
    {"device_id": "device_3", "siid": 2, "piid": 2, "value": 50},
]

# 执行批量操作（自动刷新缓存）
results = api.batch_control_devices(requests)

# 处理结果
for i, result in enumerate(results):
    if result.get("code") == 0:
        print(f"请求{i+1}成功")
    else:
        print(f"请求{i+1}失败: {result.get('message')}")
```

### 高频批量操作

对于需要频繁批量控制设备的场景（如动画效果、渐变调节等），可以禁用自动刷新缓存以提高性能：

```python
# 高频批量操作：禁用自动刷新
for i in range(10):
    requests = [
        {"device_id": "light_1", "siid": 2, "piid": 2, "value": i * 10},
        {"device_id": "light_2", "siid": 2, "piid": 2, "value": i * 10},
    ]
    api.batch_control_devices(requests, refresh_cache=False)

# 操作完成后手动刷新缓存
api.refresh_cache(home_id="123456")
```

### 批量获取设备状态

```python
def get_all_devices_in_home(api, home_id):
    """获取家庭下所有设备的详细状态"""
    devices = api.get_devices(home_id=home_id)
    
    device_details = []
    for device in devices:
        # 获取设备详细信息
        detail = api.get_device(device_id=device.did)
        if detail:
            device_details.append(detail)
    
    return device_details
```

## 缓存管理

### 自动刷新缓存

SDK 在设备控制操作后会自动刷新缓存，这是标准做法，确保下次获取的是最新状态：

```python
# 控制设备后自动刷新缓存（默认行为）
api.control_device("device_123", 2, 1, True)

# 下次获取设备状态时会得到最新数据
device = api.get_device("device_123")
```

### 禁用自动刷新

对于高频操作场景，可以禁用自动刷新以提高性能：

```python
# 连续调节亮度（禁用自动刷新）
for brightness in range(0, 100, 10):
    api.control_device(
        "light_123", 
        2, 
        2, 
        brightness, 
        refresh_cache=False
    )

# 操作完成后手动刷新
api.refresh_cache(home_id="123456")
```

### 主动刷新缓存

```python
# 刷新特定家庭的缓存
api.refresh_cache(home_id="123456")

# 刷新当前用户的所有缓存
api.refresh_cache()

# 清空所有缓存（谨慎使用）
api.clear_all_cache()
```

### 缓存刷新最佳实践

1. **单次操作**：使用默认的自动刷新（`refresh_cache=True`）
2. **高频操作**：禁用自动刷新（`refresh_cache=False`），操作完成后手动刷新
3. **批量操作**：使用 `batch_control_devices()` 并保持默认的自动刷新
4. **定时任务**：在任务开始前调用 `refresh_cache()` 确保数据最新

```python
# 示例：定时调节灯光
import schedule

def adjust_lights():
    # 任务开始前刷新缓存
    api.refresh_cache(home_id="123456")
    
    # 获取设备列表
    devices = api.get_devices(home_id="123456")
    
    # 批量控制（自动刷新缓存）
    requests = [
        {"device_id": d.did, "siid": 2, "piid": 1, "value": True}
        for d in devices if d.model.startswith("yeelink")
    ]
    api.batch_control_devices(requests)

schedule.every().hour.do(adjust_lights)
```

### 自定义统计

```python
def analyze_devices(api, home_id):
    """分析设备使用情况"""
    devices = api.get_devices(home_id=home_id)
    
    # 按房间统计
    by_room = {}
    for device in devices:
        room_id = device.room_id or "未分配"
        by_room[room_id] = by_room.get(room_id, 0) + 1
    
    # 按状态统计
    online = sum(1 for d in devices if d.status == 1)
    offline = len(devices) - online
    
    # 按品牌统计（从型号推断）
    by_brand = {}
    for device in devices:
        brand = device.model.split('.')[0]
        by_brand[brand] = by_brand.get(brand, 0) + 1
    
    return {
        "total": len(devices),
        "online": online,
        "offline": offline,
        "by_room": by_room,
        "by_brand": by_brand,
    }

# 使用
stats = analyze_devices(api, home_id)
print(f"设备分析: {stats}")
```

## Redis缓存

### 为什么使用Redis？

**适用场景**：
- 多进程应用（如Gunicorn多worker）
- 多服务器部署
- 需要共享缓存的场景

### 配置Redis

**方式一：配置文件**
```toml
# config.toml
[redis]
enabled = true
host = "localhost"
port = 6379
db = 0
password = ""  # 如果有密码
```

**方式二：环境变量**
```bash
export MIJIA_REDIS_ENABLED=true
export MIJIA_REDIS_HOST=localhost
export MIJIA_REDIS_PORT=6379
```

**方式三：代码配置**
```python
from mijiaAPI_V2 import create_api_client
from mijiaAPI_V2.core.config import ConfigManager
import redis

# 创建Redis客户端
redis_client = redis.Redis(
    host='localhost',
    port=6379,
    db=0,
    decode_responses=True
)

# 创建API客户端时传入Redis客户端
api = create_api_client(
    credential=credential,
    redis_client=redis_client
)
```

### Redis缓存的优势

```python
# 进程A
api_a = create_api_client(credential, redis_client=redis_client)
devices = api_a.get_devices(home_id)  # 从API获取，缓存到Redis

# 进程B（可以在不同服务器上）
api_b = create_api_client(credential, redis_client=redis_client)
devices = api_b.get_devices(home_id)  # 直接从Redis获取，无需调用API
```

## 错误重试

### 自动重试机制

HTTP客户端内置了自动重试机制：

```python
# 默认配置
max_retries = 3
retry_delay = 1  # 秒

# 重试策略：指数退避
# 第1次重试：等待1秒
# 第2次重试：等待2秒
# 第3次重试：等待4秒
```

### 自定义重试策略

```python
from mijiaAPI_V2.core.config import ConfigManager

config = ConfigManager()
config.set("NETWORK_MAX_RETRIES", 5)
config.set("NETWORK_RETRY_DELAY", 2)

api = create_api_client(credential)
```

### 手动重试

```python
from tenacity import retry, stop_after_attempt, wait_exponential

@retry(
    stop=stop_after_attempt(3),
    wait=wait_exponential(multiplier=1, min=1, max=10)
)
def control_device_with_retry(api, device_id, siid, piid, value):
    """带重试的设备控制"""
    return api.control_device(device_id, siid, piid, value)

# 使用
try:
    success = control_device_with_retry(api, "device_id", 2, 1, True)
    print(f"控制成功: {success}")
except Exception as e:
    print(f"重试3次后仍然失败: {e}")
```

## 性能优化

### 1. 使用缓存

```python
# 频繁访问的数据应该缓存
devices = api.get_devices(home_id)  # 第一次从API获取
devices = api.get_devices(home_id)  # 第二次从缓存获取（快）
```

### 2. 批量操作

```python
# 不好：逐个控制设备
for device_id in device_ids:
    api.control_device(device_id, 2, 1, True)

# 好：批量控制
requests = [
    {"device_id": did, "siid": 2, "piid": 1, "value": True}
    for did in device_ids
]
api.batch_control_devices(requests)
```

### 3. 异步并发

```python
# 不好：顺序执行
devices_1 = await api.get_devices(home_id_1)
devices_2 = await api.get_devices(home_id_2)
devices_3 = await api.get_devices(home_id_3)

# 好：并发执行
devices_1, devices_2, devices_3 = await asyncio.gather(
    api.get_devices(home_id_1),
    api.get_devices(home_id_2),
    api.get_devices(home_id_3),
)
```

### 4. 连接复用

```python
# 不好：每次创建新客户端
for _ in range(100):
    api = create_api_client(credential)
    api.get_devices(home_id)

# 好：复用客户端
api = create_api_client(credential)
for _ in range(100):
    api.get_devices(home_id)
```
