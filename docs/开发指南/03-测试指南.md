# 测试指南

本文档详细说明如何编写和运行测试。

## 测试框架

项目使用以下测试工具：

- **pytest** - 测试框架
- **pytest-cov** - 覆盖率报告
- **pytest-mock** - Mock支持
- **pytest-asyncio** - 异步测试支持
- **hypothesis** - 属性测试

---

## 运行测试

### 基本命令

```bash
# 运行所有测试
uv run pytest

# 运行单元测试
uv run pytest tests/unit/

# 运行集成测试
uv run pytest tests/integration/

# 运行特定文件
uv run pytest tests/unit/test_device_service.py

# 运行特定测试
uv run pytest tests/unit/test_device_service.py::test_get_devices
```

### 测试选项

```bash
# 详细输出
uv run pytest -v

# 显示print输出
uv run pytest -s

# 失败时停止
uv run pytest -x

# 只运行失败的测试
uv run pytest --lf

# 并行运行（需要pytest-xdist）
uv run pytest -n auto
```

### 覆盖率报告

```bash
# 生成覆盖率报告
uv run pytest --cov=mijiaAPI_V2

# 生成HTML报告
uv run pytest --cov=mijiaAPI_V2 --cov-report=html

# 显示缺失的行
uv run pytest --cov=mijiaAPI_V2 --cov-report=term-missing

# 设置最小覆盖率
uv run pytest --cov=mijiaAPI_V2 --cov-fail-under=80
```

### 使用Makefile

```bash
# 运行测试
make test

# 测试覆盖率
make test-cov

# 查看HTML报告
open htmlcov/index.html
```

## 单元测试

### 基本示例

```python
# tests/unit/services/test_device_service.py

import pytest
from mijiaAPI_V2.services.device_service import DeviceService
from mijiaAPI_V2.domain.models import Device, Credential
from mijiaAPI_V2.domain.exceptions import DeviceNotFoundError


class TestDeviceService:
    """设备服务测试"""
    
    def test_get_devices_returns_list(self, device_service, credential):
        """测试获取设备列表返回列表"""
        # Arrange
        home_id = "123"
        
        # Act
        devices = device_service.get_devices(home_id, credential)
        
        # Assert
        assert isinstance(devices, list)
        assert len(devices) > 0
        assert all(isinstance(d, Device) for d in devices)
    
    def test_get_device_by_id_returns_device(self, device_service, credential):
        """测试根据ID获取设备返回设备对象"""
        # Arrange
        device_id = "device_123"
        
        # Act
        device = device_service.get_device_by_id(device_id, credential)
        
        # Assert
        assert device is not None
        assert device.did == device_id
    
    def test_get_device_by_id_raises_error_when_not_found(
        self, device_service, credential
    ):
        """测试获取不存在的设备抛出异常"""
        # Arrange
        device_id = "non_existent"
        
        # Act & Assert
        with pytest.raises(DeviceNotFoundError):
            device_service.get_device_by_id(device_id, credential)
```

## 异步测试

### 基本异步测试

```python
import pytest
import asyncio


@pytest.mark.asyncio
async def test_async_get_devices():
    """测试异步获取设备"""
    # Arrange
    api = create_async_api_client(credential)
    
    # Act
    devices = await api.get_devices(home_id="123")
    
    # Assert
    assert isinstance(devices, list)


@pytest.mark.asyncio
async def test_concurrent_requests():
    """测试并发请求"""
    api = create_async_api_client(credential)
    
    # Act
    results = await asyncio.gather(
        api.get_devices("home_1"),
        api.get_devices("home_2"),
        api.get_devices("home_3"),
    )
    
    # Assert
    assert len(results) == 3
    assert all(isinstance(r, list) for r in results)
```

```python
# tests/integration/test_api_client.py

import pytest
from mijiaAPI_V2 import create_api_client, create_auth_service


class TestAPIIntegration:
    """API集成测试"""
    
    @pytest.fixture(scope="class")
    def api_client(self):
        """创建API客户端"""
        auth_service = create_auth_service()
        credential = auth_service.load_credential()
        return create_api_client(credential)
    
    def test_full_workflow(self, api_client):
        """测试完整工作流"""
        # 1. 获取家庭列表
        homes = api_client.get_homes()
        assert len(homes) > 0
        
        # 2. 获取设备列表
        home_id = homes[0].id
        devices = api_client.get_devices(home_id)
        assert len(devices) > 0
        
        # 3. 获取智能列表
        scenes = api_client.get_scenes(home_id)
        assert isinstance(scenes, list)
    
    def test_device_control_workflow(self, api_client):
        """测试设备控制流程"""
        # 1. 获取设备
        homes = api_client.get_homes()
        devices = api_client.get_devices(homes[0].id)
        
        # 2. 找到在线设备
        online_device = next(
            (d for d in devices if d.status == 1), 
            None
        )
        
        if online_device:
            # 3. 控制设备
            success = api_client.control_device(
                device_id=online_device.did,
                siid=2,
                piid=1,
                value=True
            )
            assert success is True
```

## 属性测试

### 使用Hypothesis

```python
from hypothesis import given, strategies as st


@given(st.text(min_size=1, max_size=50))
def test_device_name_validation(name):
    """测试设备名称验证（属性测试）"""
    device = Device(
        did="123",
        name=name,
        model="test",
        home_id="home",
        status=1
    )
    assert device.name == name
    assert len(device.name) <= 50


@given(st.integers(min_value=0, max_value=1))
def test_device_status_validation(status):
    """测试设备状态验证"""
    device = Device(
        did="123",
        name="test",
        model="test",
        home_id="home",
        status=status
    )
    assert device.status in [0, 1]


@given(
    st.lists(
        st.builds(
            Device,
            did=st.text(min_size=1),
            name=st.text(min_size=1),
            model=st.text(min_size=1),
            home_id=st.text(min_size=1),
            status=st.integers(min_value=0, max_value=1)
        ),
        min_size=0,
        max_size=100
    )
)
def test_device_list_operations(devices):
    """测试设备列表操作"""
    # 筛选在线设备
    online = [d for d in devices if d.status == 1]
    assert all(d.status == 1 for d in online)
    
    # 按名称排序
    sorted_devices = sorted(devices, key=lambda d: d.name)
    assert len(sorted_devices) == len(devices)
```

## 测试最佳实践

### 1. AAA模式

```python
def test_example():
    # Arrange（准备）- 设置测试数据和环境
    device = Device(...)
    service = DeviceService(...)
    
    # Act（执行）- 执行被测试的操作
    result = service.process_device(device)
    
    # Assert（断言）- 验证结果
    assert result is True
```

### 2. 一个测试一个断言

```python
# ✅ 好：每个测试只验证一件事
def test_device_is_online():
    device = Device(status=1)
    assert device.is_online() is True

def test_device_is_offline():
    device = Device(status=0)
    assert device.is_online() is False

# ❌ 不好：一个测试验证多件事
def test_device_status():
    device1 = Device(status=1)
    device2 = Device(status=0)
    assert device1.is_online() is True
    assert device2.is_online() is False
```

### 3. 测试命名清晰

```python
# ✅ 好：清晰的测试名称
def test_get_devices_returns_empty_list_when_home_has_no_devices():
    pass

def test_control_device_raises_error_when_device_offline():
    pass

# ❌ 不好：模糊的测试名称
def test_devices():
    pass

def test_control():
    pass
```

### 4. 使用Fixtures减少重复

```python
# ✅ 好：使用fixture
@pytest.fixture
def sample_device():
    return Device(did="123", name="test", ...)

def test_device_online(sample_device):
    assert sample_device.is_online()

def test_device_name(sample_device):
    assert sample_device.name == "test"

# ❌ 不好：重复创建对象
def test_device_online():
    device = Device(did="123", name="test", ...)
    assert device.is_online()

def test_device_name():
    device = Device(did="123", name="test", ...)
    assert device.name == "test"
```

### 5. 测试边界条件

```python
def test_validate_brightness():
    # 正常值
    assert validate_brightness(50) is True
    
    # 边界值
    assert validate_brightness(0) is True
    assert validate_brightness(100) is True
    
    # 无效值
    with pytest.raises(ValidationError):
        validate_brightness(-1)
    with pytest.raises(ValidationError):
        validate_brightness(101)
```
